IF EXISTS (SELECT name FROM sys.databases WHERE name = N'campo')
BEGIN
    ALTER DATABASE campo SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
    DROP DATABASE campo;
END
GO

CREATE DATABASE campo;
GO

USE campo;
GO

CREATE TABLE Roles (
    Id INT PRIMARY KEY IDENTITY(1,1),
    Nombre VARCHAR(50) NOT NULL,
    Activo BIT DEFAULT 1 NOT NULL
);

CREATE TABLE Permisos (
    Id INT PRIMARY KEY IDENTITY(1,1),
    Nombre VARCHAR(50) NOT NULL,
    Descripcion VARCHAR(255)
);

CREATE TABLE Usuario (
    Id INT PRIMARY KEY IDENTITY(1,1),
    Nombre VARCHAR(100) NOT NULL UNIQUE,
    Contraseña VARCHAR(255) NOT NULL,
    Activo BIT DEFAULT 1 NOT NULL,
    DVH BIGINT DEFAULT 0 NOT NULL
);

CREATE TABLE Tabla_DVV (
    Nombre_Tabla VARCHAR(50) PRIMARY KEY,
    Valor_DVV BIGINT DEFAULT 0 NOT NULL
);

CREATE TABLE Usuario_Historial (
    Id_Historial INT PRIMARY KEY IDENTITY(1,1),
    Id_Usuario_Afectado INT NOT NULL,
    Nombre_Anterior VARCHAR(100),
    Contraseña_Anterior VARCHAR(255),
    Activo_Anterior BIT,
    Id_Usuario_Responsable INT NOT NULL,
    Fecha_Cambio DATETIME DEFAULT GETDATE(),
    Tipo_Operacion VARCHAR(50)
);

CREATE TABLE Idioma (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(50) NOT NULL UNIQUE,
    PorDefecto BIT NOT NULL DEFAULT 0
);

CREATE TABLE Roles_Permisos (
    RolId INT NOT NULL,
    PermisoId INT NOT NULL, 
    PRIMARY KEY (RolId, PermisoId),
    FOREIGN KEY (RolId) REFERENCES Roles(Id),
    FOREIGN KEY (PermisoId) REFERENCES Permisos(Id)
);

CREATE TABLE Usuario_Roles (
    UsuarioId INT NOT NULL,
    RolId INT NOT NULL,
    PRIMARY KEY (UsuarioId, RolId),
    FOREIGN KEY (UsuarioId) REFERENCES Usuario(Id),
    FOREIGN KEY (RolId) REFERENCES Roles(Id)
);

CREATE TABLE Bitacora (
    Id INT PRIMARY KEY IDENTITY(1,1),
    Id_Usuario INT NOT NULL,
    Accion VARCHAR(100) NOT NULL,
    Detalle VARCHAR(255),
    Fecha DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (Id_Usuario) REFERENCES Usuario(Id)
);

CREATE TABLE Traduccion (
    Idioma_Id INT FOREIGN KEY REFERENCES Idioma(Id),
    Clave_Control VARCHAR(100) NOT NULL,
    Texto NVARCHAR(MAX) NOT NULL,
    PRIMARY KEY (Idioma_Id, Clave_Control)
);
